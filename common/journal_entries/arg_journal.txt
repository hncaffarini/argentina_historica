arg_mision_organizacion_nacional = {
	title = "La Crisis de la Confederación"
	icon = "gfx/interface/icons/event_icons/civil_war.dds"
	
	# Aparece cuando ocurre el evento del Pronunciamiento
	trigger = {
		has_variable = arg_mision_caseros_activa
		tag = ARG
	}

	# Objetivo: Mantener el orden o colapsar
	status_proxies = {
		# Esto crea una barra que se llena con el radicalismo
		variable = arg_progreso_caida_rosas
	}

	# Condición de fracaso (Histórica): Si el radicalismo en el Litoral es muy alto
	# o si pasa demasiado tiempo sin reformas.
	timeout_effect = {
		trigger_event = { id = arg_batalla_caseros.1 }
	}

	# Condición de éxito (Alternativa): Sobrevivir 2 años al asedio político
	complete = {
		years_passed_since_entry = 2
		ig:ig_landowners = { approval > 10 }
	}

	on_complete = {
		# Rosas sobrevive y se consolida
		add_prestige = 100
	}
}

arg_mision_gobernar_es_poblar = {
	title = "Gobernar es Poblar"
	icon = "gfx/interface/icons/event_icons/population.dds"
	
	transferable = yes

	# Se activa con la unificación nacional
	trigger = {
		has_variable = arg_mitre_lanzado
		tag = ARG
	}

	# Objetivos para completar la misión (Simulando 1862-1880)
	is_complete = {
		custom_description = {
			text = "arg_objetivo_poblacion_desc"
			country_population >= 3000000
		}
		custom_description = {
			text = "arg_objetivo_alfabetizacion_desc"
			average_literacy >= 0.25
		}
		custom_description = {
			text = "arg_objetivo_ferrocarril_desc"
			has_technology = railways
			any_scope_state = {
				has_building_with_level = {
					building = railway
					level >= 1
				}
			}
		}
	}

	on_complete = {
		add_prestige = 100
		add_modifier = { name = arg_granero_del_mundo months = 240 }
		trigger_event = { id = arg_fin_mision_poblar.1 }
	}
}
